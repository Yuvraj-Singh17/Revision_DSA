import java.util.*;

class Main {

    static void findSubsets(int[] arr, int n, int k) {
        Arrays.sort(arr);
        backtrack(arr, k, 0, new ArrayList<>());
    }

    static void backtrack(int[] arr, int target, int index, ArrayList<Integer> current) {

        if (target == 0) {
            for (int num : current) {
                System.out.print(num + " ");
            }
            System.out.println();
            return;
        }

        for (int i = index; i < arr.length; i++) {
            if (i > index && arr[i] == arr[i - 1]) continue;
            if (arr[i] > target) break;
            current.add(arr[i]);
            backtrack(arr, target - arr[i], i + 1, current);
            current.remove(current.size() - 1);
        }
    }

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int k = sc.nextInt();

        int[] arr = new int[n];
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }

        findSubsets(arr, n, k);
        sc.close();
    }
}
